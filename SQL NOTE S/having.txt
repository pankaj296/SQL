select sales_date ,product_id  ,sum(total_amount)
from sales 
where sales_date BETWEEN '01-jan-2015' AND '31-jan-2015'
GROUP BY sales_date , product_id
HAVING sum(total_amount)>100;

--3. Produce a list which shows maximum sales volume broken by sales date and order
--number from SALES table.

SELECT *  FROM sales;

select sales_date ,product_id ,customer_id ,sum(total_amount)
from sales 
where sales_date BETWEEN '01-jan-2015' AND '31-jan-2015'
GROUP BY sales_date , product_id,customer_id
ORDER BY  sales_date;


--4. Produce a list which shows minimum sales volume broken by sales date from SALES table
--and minimum sales volume is less than 100. (Hint: use HAVING clause)



SELECT sales_date, MIN(sales_amount)
 FROM sales
GROUP BY sales_date
 HAVING MIN(sales_amount)< 100
