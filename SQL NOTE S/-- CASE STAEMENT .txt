-- CASE STAEMENT 

SELECT sales_date ,sales_amount,
CASE
WHEN total_amount <= 100 THEN 'LOW'
WHEN total_amount < 1000 THEN 'MEDIUM'
ELSE 'HIGH'
END
FROM SALES;
